		class MultiplicationLayer()
			initialize(hiperparameters)
				generate filters with starting values
				use hiperparameters for that !
		
			forward(x, y)
				z = x * y
				self.x = x # must keep these around !
				self.y = y # this we CACHE too
				return z
			
			backward(dz)
				dx = self.y * dz # dz/dx * dL/dz
				dy = self.x * dz # dz/dy * dL/dz
				return [dx, dy]
